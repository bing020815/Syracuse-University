    var searchResultsTable;

    $(document).ready(function () {
        // set csrf header for ajax calls
        var csrfHeader = $("meta[name='_csrf_header']").attr(
            "content");
        var csrfToken = $("meta[name='_csrf']").attr("content");
        var headers = {};
        headers[csrfHeader] = csrfToken;
        $.ajaxSetup({
            headers : headers
        });

        searchResultsTable = $('#searchResultsTable').DataTable({
            dom: 'lrtip',
            lengthMenu : [ [ 10, 25, 50, 75, 100, -1 ], [ 10, 25, 50, 75, 100, "All" ] ],
            columns: [
                {data: 'cohortYear'},
                {data: 'orgName'},
                {data: 'conferenceName'},
                {data: 'sportDesc'},
                {data: 'state'},
                {data: 'gsr', visible : true},
                {data: 'fgr'},
                {data: null,
                    visible : true,
                    render: function(data, type, row) {
                        return '<a href="'+data.gsrReportLink+'?v=1579241526641" target="_blank"><span class="btn glyphicon glyphicon-file"></span></a>';
                    }},
                {data: null,
                    render: function(data, type, row) {
                        return '<a href="'+data.fgrReportLink+'?v=1579241526641" target="_blank"><span class="btn glyphicon glyphicon-file"></span></a>';
                    }}
            ]
        });
    })

    function search() {
        $('#pleaseWaitDialog').modal('show');
        $.ajax({
            type: "POST",
            data: $('#gsrSearchForm').serialize(),
            dataType: "json",
            success: function(json) {
                $('#pleaseWaitDialog').modal('hide');
                hideErrorMsg();
                updateSearchResultsHeader();
                updateSearchResultsTable (json);
            },
            error: function(xhr, error, thrown) {
                $('#pleaseWaitDialog').modal('hide');
                displayErrorMsg();
            }
        });
        return false;
    }

    function updateSearchResultsTable(json) {
        searchResultsTable.clear();
        searchResultsTable.rows.add(json);
        searchResultsTable.columns.adjust().draw();
        $('html, body').animate({
            scrollTop : $('#searchResultsPanel').offset().top - 100
        }, 'slow');
    }

    function updateSearchResultsHeader() {
        var searhParams = '';
        $('#aprSearchForm select').each(function() {
            if ($(this).val() != '') {
                if (searhParams != '') {
                    searhParams = searhParams+', ';
                }
                searhParams = searhParams+$(this).find('option:selected').text();
            }
        })
        $('#searchParams').html(searhParams);
    }

    function displayErrorMsg() {
        $('#errorMsgDiv').hide().removeClass('hidden').show('slow');
    }

    function hideErrorMsg() {
        $('#errorMsgDiv').hide('slow', function(){$(this).addClass('hidden');});
    }

</script>
